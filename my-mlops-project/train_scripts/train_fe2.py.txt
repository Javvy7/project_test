#!/usr/bin/env python
# coding: utf-8

from src.fe2.fe2 import FeatureEngineering2
import pandas as pd
import boto3

bucket_name = 'dataminds-warehouse'
s3_file_key = 'ramen-ratings.csv'
local_file_path = 'ramen-ratings.csv'

s3 = boto3.client('s3', region_name='us-east-1')
s3.download_file(bucket_name, s3_file_key, local_file_path)

df = pd.read_csv(local_file_path)

fe2 = FeatureEngineering2(df, target_column='Stars')
fe2.clean_data()
fe2.split_data()
fe2.create_preprocessor()
fe2.train_model()
fe2.evaluate()
